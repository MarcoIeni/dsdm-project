44  std::vector<int> &weights, std::vector<Edge> &edges,
45  int control_steps) {
46  std::string second_line;

48 
49  //take numbers from string and put it into vector
50  std::stringstream stream(second_line);
51  std::vector<int> values((std::istream_iterator<int>(stream)), // begin
52  (std::istream_iterator<int>())); // end
53 
54  int tail_node, head_node;
55  tail_node = values[0];
56  head_node = values[1];

66  weights.push_back(weight);

69  } else {
70  // the weight is not specified in the input file
71  do {
72  edges.push_back(Edge(tail_node, head_node));
73  if (tail_node == SOURCE) { // edge outgoing from entry
74  weights.push_back(control_steps);
75  } else {
76  weights.push_back(0);
77  }
78  } while (infile >> tail_node >> head_node);

176 bool isInPath(int v, int src, int end, int parent[]) {
177  int i = end;
178  if (v == src || v == end) {
179  return true;
180  }

182  if (v == parent[i]) {
183  return true;
184  }
185  i = parent[i];
186  }

190 void RetimingManager::dijkstra_max_delay(int src, int* dist, int* tot_delays) {
191  const weights_map weights = get(boost::edge_weight, *main_graph);
192 
193  /*
194  * Value of parent[v] for a vertex v stores parent vertex of v in
195  * shortest path tree. Parent of root (or source vertex) is -1.
196  * Whenever we find shorter path through a vertex u, we make u as
197  * parent of current vertex.
198  */
199  int parent[num_vertices(*main_graph)];
200 
201  // Create a priority queue to store vertices that are being preprocessed
202  std::priority_queue<iPair, std::vector<iPair>, std::greater<iPair> > pq;
203  const IndexMap index = boost::get(boost::vertex_index, *main_graph);
204 
205  // Insert source itself in priority queue and initialize its distance as 0.
206  pq.push(std::make_pair(0, src));
207  dist[src] = 0;
208  parent[src] = -1;
209  tot_delays[src] = delays[src];
210 
211  // Looping till priority queue becomes empty (or all distances are not finalized)
212  while (!pq.empty()) {
213  // The first vertex in pair is the minimum distance
214  // vertex, extract it from priority queue.
215  // vertex label is stored in second of pair.
216  int u = pq.top().second;
217  pq.pop();
218  typename GraphTraits::adjacency_iterator ai;
219  typename GraphTraits::adjacency_iterator ai_end;
220  typename GraphTraits::out_edge_iterator out_i, out_end;
221  typename GraphTraits::edge_descriptor e;
222  Vertex targ;
223  for (boost::tie(out_i, out_end) = boost::out_edges(u, *main_graph);
224  out_i != out_end; ++out_i) {
225  e = *out_i;
226  targ = boost::target(e, *main_graph);
227  int v = index[targ];
228  int weight = boost::get(weights, e);
229  if (dist[v] > dist[u] + weight
230  || (weight + dist[u] == dist[v]
231  && delays[v] + tot_delays[u] > tot_delays[v])) {
232  if (!isInPath(v, src, u, parent)) {
233  // Updating total delay and distance of v
234  tot_delays[v] = delays[v] + tot_delays[u];
235  dist[v] = dist[u] + weight;
236  parent[v] = u;
237  pq.push(std::make_pair(dist[v], v));
238  }
239  }
240  }
241  }
242 }

296 void RetimingManager::move_weights(std::vector<int> &pc_distances) {
297  boost::property_map<Graph, boost::vertex_index_t>::type vertex_id =
298  boost::get(boost::vertex_index, *main_graph);
299  boost::graph_traits<Graph>::edge_iterator ei, ei_end;
300  for (boost::tie(ei, ei_end) = edges(*main_graph); ei != ei_end; ++ei) {
301  int weight, rv, ru;
302 
303  // I do +1 because pc_graph adds an origin node to the main graph
304  rv = pc_distances[boost::get(vertex_id, target(*ei, *main_graph)) + 1];
305  ru = pc_distances[boost::get(vertex_id, source(*ei, *main_graph)) + 1];
306 
307  weight = get(boost::edge_weight_t(), *main_graph, *ei);
308  boost::put(boost::edge_weight_t(), *main_graph, *ei, weight + rv - ru);
309  }

312 std::vector<int> RetimingManager::get_scheduling() const {
313  std::vector<bool> visited(num_vertices(*main_graph), false);
314  const weights_map weights = get(boost::edge_weight, *main_graph);
315  typename GraphTraits::out_edge_iterator out_i, out_end;
316  typename GraphTraits::edge_descriptor e;
317  Vertex targ;
318  std::vector<int> scheduling(num_vertices(*main_graph));
319  const IndexMap index = boost::get(boost::vertex_index, *main_graph);
320 
321  // To get the scheduling I do a bfs visit
322  std::list<int> q;
323  scheduling[SOURCE] = delays[0];
324  visited[SOURCE] = true;
325  q.push_back(SOURCE);
326  while (!q.empty()) {
327  int u = q.front();
328  q.pop_front();
329  for (boost::tie(out_i, out_end) = boost::out_edges(u, *main_graph);
330  out_i != out_end; ++out_i) {
331  e = *out_i;
332  targ = boost::target(e, *main_graph);
333  int v = index[targ];
334  if (!visited[v]) {
335  visited[v] = true;
336  scheduling[v] = scheduling[u] + boost::get(weights, e);
337  q.push_back(v);
338  }
339  }
340  }
341  return scheduling;

368 bool RetimingManager::is_main_graph_cyclic_util(int v, bool visited[],
369  bool *recStack) {
370  const IndexMap index = boost::get(boost::vertex_index, *main_graph);
371  const weights_map weights = get(boost::edge_weight, *main_graph);
372 
373  if (visited[v] == false) {
374  // Mark the current node as visited and part of recursion stack
375  visited[v] = true;
376  recStack[v] = true;
377 
378  typename GraphTraits::adjacency_iterator ai;
379  typename GraphTraits::adjacency_iterator ai_end;
380  typename GraphTraits::out_edge_iterator out_i, out_end;
381  typename GraphTraits::edge_descriptor e;
382  Vertex targ;
383  for (boost::tie(out_i, out_end) = boost::out_edges(v, *main_graph);
384  out_i != out_end; ++out_i) {
385  e = *out_i;
386  targ = boost::target(e, *main_graph);
387  int t = index[targ];
388  int weight = boost::get(weights, e);
389 
390  if (weight == 0) {
391  if (!visited[t]
392  && is_main_graph_cyclic_util(t, visited, recStack))
393  return true;
394  else if (recStack[t])
395  return true;
396  }
397  }
398  }
399  recStack[v] = false; // remove the vertex from recursion stack
400  return false;
401 }
